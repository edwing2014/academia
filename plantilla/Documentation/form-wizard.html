Usage

 $('#stepy_form').validate({
        errorPlacement: function (error, element) {
            $('#stepy_form div.stepy-error').append(error);
        },
        rules: {
            'name': 'required',
            'email': 'required',
        },
        messages: {
            'name': {
                required: 'Name field is required!'
            },
            'email': {
                required: 'Email field is requerid!'
            },
        }
    });
  });
